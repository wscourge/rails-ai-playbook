# frozen_string_literal: true

require "net/http"
require "json"

# Send transactional SMS via Brevo API
#
# Uses the Brevo Transactional SMS endpoint:
# POST https://api.brevo.com/v3/transactionalSMS/sms
#
# @example
#   SmsSender.send_sms(
#     to: "+48123456789",
#     content: "Your verification code is 123456"
#   )
#
# @see https://developers.brevo.com/reference/sendtransacsms
#
class SmsSender
  BREVO_SMS_ENDPOINT = "https://api.brevo.com/v3/transactionalSMS/sms"

  class << self
    # Send an SMS message
    #
    # @param to [String] Recipient phone number in E.164 format (e.g. "+48123456789")
    # @param content [String] SMS message body
    # @param sender [String, nil] Override sender name (defaults to config)
    # @return [Hash] Brevo API response body
    # @raise [SmsSender::Error] on failure
    def send_sms(to:, content:, sender: nil)
      config = Config.sms

      unless config.active?
        Rails.logger.info("[SmsSender] SMS not active (env=#{Rails.env}). Would send to #{to}: #{content}")
        return { status: "skipped", message: "SMS not active" }
      end

      payload = {
        type: "transactional",
        sender: sender || config.sender_name,
        recipient: to,
        content: content,
      }

      response = post_to_brevo(payload, config.brevo_api_key)

      unless response.is_a?(Net::HTTPSuccess)
        body = safe_parse(response.body)
        raise Error, "Brevo SMS API error (#{response.code}): #{body}"
      end

      safe_parse(response.body)
    end

    private

    def post_to_brevo(payload, api_key)
      uri = URI(BREVO_SMS_ENDPOINT)
      request = Net::HTTP::Post.new(uri)
      request["accept"] = "application/json"
      request["content-type"] = "application/json"
      request["api-key"] = api_key
      request.body = payload.to_json

      Net::HTTP.start(uri.hostname, uri.port, use_ssl: true) do |http|
        http.request(request)
      end
    end

    def safe_parse(body)
      JSON.parse(body)
    rescue JSON::ParserError
      { "raw" => body }
    end
  end

  class Error < StandardError; end
end
